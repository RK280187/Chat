// services/astrology-service/src/kundli/calculator.ts import { JPLEphemeris } from '@jpl/ephemeris'; import { getNakshatra, calculateLagna } from './vedic-math'; interface BirthData { birthTime: Date; latitude: number; longitude: number; timezone: number; } export async function calculateKundli(birthData: BirthData) { const ephemeris = new JPLEphemeris(); // Calculate planetary positions with JPL NASA precision const planetPositions = await ephemeris.calculate({ date: birthData.birthTime, latitude: birthData.latitude, longitude: birthData.longitude, timezone: birthData.timezone, ayanamsa: 23.11, // Lahiri ayanamsa }); // Detect Nakshatra from Moon position const moonDegree = planetPositions.moon.degree; const nakshatra = getNakshatra(moonDegree); // Calculate Lagna (Ascendant) const lagna = calculateLagna(planetPositions, birthData); // Calculate all 16 divisional charts (Vargas) const vargas = [ calculateVarga('rashi', planetPositions), calculateVarga('navamsa', planetPositions), calculateVarga('dasamsa', planetPositions), // ... 13 more divisional charts ]; return { planetPositions, nakshatra, lagna, vargas, doshas: detectDoshas(planetPositions), yogas: detectYogas(planetPositions), shadbala: calculateShadbala(planetPositions), dashaSystem: calculateVimshottariDasha(birthData.birthTime), timestamp: new Date() }; }
// services/astrology-service/src/doshas/detector.ts export async function detectDoshas(planets: PlanetaryPositions) { const doshas = { manglik: detectManglikDosha(planets), kaalSarp: detectKaalSarpDosha(planets), pitra: detectPitraDosha(planets), nadi: detectNadiDosha(planets), }; return doshas; } function detectManglikDosha(planets: PlanetaryPositions): boolean { // Mars in 1st, 4th, 7th, 8th, 12th house = Manglik const criticalHouses = [1, 4, 7, 8, 12]; return criticalHouses.includes(planets.mars.house); } function detectKaalSarpDosha(planets: PlanetaryPositions): boolean { // All planets between Rahu and Ketu = Kaal Sarp const rahuDegree = planets.rahu.degree; const ketuDegree = planets.ketu.degree; const allPlanets = [ planets.sun, planets.moon, planets.mercury, planets.venus, planets.mars, planets.jupiter, planets.saturn ]; return allPlanets.every(p => (p.degree > rahuDegree && p.degree < ketuDegree) || (rahuDegree > ketuDegree && (p.degree > rahuDegree || p.degree < ketuDegree)) ); }
# services/ai-service/src/predictions/ensemble.py import torch import numpy as np from transformers import AutoModel import xgboost as xgb class EnsemblePredictionEngine: def __init__(self): self.transformer = AutoModel.from_pretrained( 'nakshatra-ai/transformer-v1' ) self.lstm = load_model('lstm_model.h5') self.xgboost = xgb.Booster( model_file='xgboost_model.bin' ) async def predict(self, chart_data): # Multi-model ensemble prediction predictions = { 'transformer': self.transformer.predict(chart_data), 'lstm': self.lstm.predict(chart_data), 'xgboost': self.xgboost.predict(chart_data), 'rules': self.rules_engine(chart_data), } # Weighted ensemble (40% + 30% + 20% + 10%) final = ( 0.40 * predictions['transformer'] + 0.30 * predictions['lstm'] + 0.20 * predictions['xgboost'] + 0.10 * predictions['rules'] ) confidence = np.std(predictions.values()) return { 'prediction': final, 'confidence': confidence, 'models': predictions, }
