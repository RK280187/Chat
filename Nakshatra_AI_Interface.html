// services/astrology-service/src/kundli/calculator.ts

export const calculateKundli = async (birthData: BirthData) => {
  // JPL Ephemeris calculation
  const planetPositions = await jpLEphemeris.calculate({
    date: birthData.birthTime,
    latitude: birthData.latitude,
    longitude: birthData.longitude,
    timezone: birthData.timezone,
    ayanamsa: 23.11 // Lahiri ayanamsa
  });

  // Nakshatra detection
  const moonDegree = planetPositions.moon.degree;
  const nakshatra = getNakshatra(moonDegree);

  // Lagna calculation
  const lagna = calculateLagna(planetPositions, birthData);

  // Divisional charts
  const vargas = calculateAllDivisionalCharts(planetPositions);

  // Return complete Kundli
  return {
    planetPositions,
    nakshatra,
    lagna,
    vargas,
    doshas: detectDoshas(planetPositions),
    yogas: detectYogas(planetPositions),
    timing: new Date().toISOString()
  };
};
